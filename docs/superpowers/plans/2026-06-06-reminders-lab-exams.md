# Reminders and Lab Exam Events Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build persistent lab/exam event management, unified reminders, an Apple Reminders-style homepage card, and an SMS scheduler stub.

**Architecture:** Add a unified backend `reminders` module as the source of truth for reminder display, completion, homepage summaries, and scheduling. Add a `lab-exam-events` module that stores event-specific fields and synchronizes exactly one linked reminder per event through service-layer transactions. The Vue frontend consumes the same reminder APIs from `/reminders`, `/exams`, and the homepage card.

**Tech Stack:** TypeScript strict mode, Vue 3 `<script setup>`, TanStack Query, Vue Router, Express 5, Drizzle ORM, PostgreSQL, Zod, Vitest, Supertest, pnpm workspaces.

export function createReminderRepository(db: Database): ReminderRepository {
  return {
    async listByUserId(userId: string, filters: ReminderFilters = {}) {
      const conditions = [eq(reminders.userId, userId), isNull(reminders.deletedAt)];
      if (filters.status) conditions.push(eq(reminders.status, filters.status));
      if (filters.category) conditions.push(eq(reminders.category, filters.category));
      if (filters.showOnHome !== undefined) conditions.push(eq(reminders.showOnHome, filters.showOnHome));
      const rows = await db.select().from(reminders).where(and(...conditions)).orderBy(asc(reminders.dueAt));
      return rows
        .map(toReminderDto)
        .filter((row) => (filters.includeCompleted ? true : row.status !== "done"))
        .filter((row) => {
          if (!filters.range) return true;
          const now = new Date();
          if (filters.range === "overdue") return row.dueAt < now && row.status !== "done";
          const end = new Date(now);
          end.setHours(filters.range === "today" ? 24 : 24 * 7, 0, 0, 0);
          return row.dueAt >= now && row.dueAt <= end;
        });
    },
    async findById(userId, id) {
      const [row] = await db
        .select()
        .from(reminders)
        .where(and(eq(reminders.userId, userId), eq(reminders.id, id), isNull(reminders.deletedAt)));
      return row ? toReminderDto(row) : null;
    },
    async create(userId, input: ReminderInput) {
      const [row] = await db.insert(reminders).values({ userId, ...input }).returning();
      return toReminderDto(row);
    },
    async update(userId, id, input: ReminderUpdateInput) {
      const [row] = await db
        .update(reminders)
        .set({ ...input, updatedAt: new Date() })
        .where(and(eq(reminders.userId, userId), eq(reminders.id, id), isNull(reminders.deletedAt)))
        .returning();
      return row ? toReminderDto(row) : null;
    },
    async softDelete(userId, id) {
      const [row] = await db
        .update(reminders)
        .set({ deletedAt: new Date(), updatedAt: new Date() })
        .where(and(eq(reminders.userId, userId), eq(reminders.id, id), isNull(reminders.deletedAt)))
        .returning({ id: reminders.id });
      return Boolean(row);
    },

        .from(reminders)
        .where(and(eq(reminders.smsEnabled, true), isNull(reminders.deletedAt), lte(reminders.dueAt, now)));

import { HttpError } from "../../lib/http-error.js";
import type { ReminderInput, ReminderRepository, ReminderUpdateInput } from "./reminders.types.js";

export function createRemindersService(repository: ReminderRepository) {
  async function updateOrThrow(userId: string, id: string, input: ReminderUpdateInput) {
    const reminder = await repository.update(userId, id, input);
    if (!reminder) throw new HttpError(404, "Reminder not found");
    return reminder;
  }

  return {
    list(userId: string, filters = {}) {
      return repository.listByUserId(userId, filters);
    },
    async home(userId: string) {
      const reminders = await repository.listByUserId(userId, { showOnHome: true, includeCompleted: false });
      const active = reminders.filter((reminder) => reminder.status !== "cancelled" && reminder.status !== "done");
      return { reminders: active.slice(0, 3), pendingCount: active.length };
    },
    createCustom(userId: string, input: Omit<ReminderInput, "sourceType" | "sourceId">) {
      return repository.create(userId, { ...input, sourceType: "custom", sourceId: null });
    },
    async update(userId: string, id: string, input: ReminderUpdateInput) {
      return updateOrThrow(userId, id, input);
    },
    async setCompleted(userId: string, id: string, completed: boolean) {
      return updateOrThrow(userId, id, {
        status: completed ? "done" : "pending",
        completedAt: completed ? new Date() : null,
        snoozedUntil: completed ? null : undefined
      });
    },
    async snooze(userId: string, id: string, snoozedUntil: Date) {
      return updateOrThrow(userId, id, { status: "snoozed", snoozedUntil });
    },
    async duplicate(userId: string, id: string) {
      const existing = await repository.findById(userId, id);
      if (!existing) throw new HttpError(404, "Reminder not found");
      return repository.create(userId, {
        title: existing.title,
        category: existing.category,
        priority: existing.priority,
        startAt: existing.startAt,
        dueAt: existing.dueAt,
        location: existing.location,
        notes: existing.notes,
        sourceType: "custom",
        sourceId: null,
        reminderOffsets: existing.reminderOffsets,
        smsEnabled: existing.smsEnabled,
        showOnHome: existing.showOnHome
      });
    },
    async delete(userId: string, id: string) {
      const deleted = await repository.softDelete(userId, id);
      if (!deleted) throw new HttpError(404, "Reminder not found");
    }
  };
}

function getRouteId(id: string | string[] | undefined): string {
  if (typeof id !== "string") throw new HttpError(400, "Reminder id is required");
  if (!/^[0-9a-f]{8}-[0-9a-f]{4}-[1-5][0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i.test(id)) {
    throw new HttpError(400, "Reminder id must be a valid UUID");
  }
  return id;
}

export function createRemindersRouter(dependencies: Dependencies): Router {
  const router = Router();
  const requireAuth = authenticate(dependencies.authRepository);

  router.get("/", requireAuth, async (req, res, next) => {
    try {
      const filters = listReminderQuerySchema.parse(req.query);
      const reminders = await dependencies.remindersService.list(req.userId ?? "", filters);
      res.json({ reminders });
    } catch (error) {
      next(error);
    }
  });

  router.get("/home", requireAuth, async (req, res, next) => {
    try {
      res.json(await dependencies.remindersService.home(req.userId ?? ""));
    } catch (error) {
      next(error);
    }
  });

  router.post("/", requireAuth, async (req, res, next) => {
    try {
      const input = createReminderSchema.parse(req.body);
      const reminder = await dependencies.remindersService.createCustom(req.userId ?? "", input);
      res.status(201).json({ reminder });
    } catch (error) {
      next(error);
    }
  });

  router.put("/:id", requireAuth, async (req, res, next) => {
    try {
      const reminder = await dependencies.remindersService.update(req.userId ?? "", getRouteId(req.params.id), updateReminderSchema.parse(req.body));
      res.json({ reminder });
    } catch (error) {
      next(error);
    }
  });

  router.patch("/:id/complete", requireAuth, async (req, res, next) => {
    try {
      const input = completeReminderSchema.parse(req.body);
      const reminder = await dependencies.remindersService.setCompleted(req.userId ?? "", getRouteId(req.params.id), input.completed);
      res.json({ reminder });
    } catch (error) {
      next(error);
    }
  });

  router.patch("/:id/snooze", requireAuth, async (req, res, next) => {
    try {
      const input = snoozeReminderSchema.parse(req.body);
      const reminder = await dependencies.remindersService.snooze(req.userId ?? "", getRouteId(req.params.id), input.snoozedUntil);
      res.json({ reminder });
    } catch (error) {
      next(error);
    }

    } catch (error) {
      next(error);

      res.status(204).send();
    } catch (error) {
      next(error);
    }
  });

  return router;
}

import type { ReminderRepository } from "./reminders.types.js";
import type { SmsAdapter } from "./sms-adapter.js";

interface SchedulerDependencies {
  now: Date;
  repository: ReminderRepository;
  smsAdapter: SmsAdapter;
}

export interface SchedulerResult {
  scanned: number;
  sent: number;
  skipped: number;
  failed: number;
}

function scheduledTimeForOffset(dueAt: Date, offsetMinutes: number): Date {
  return new Date(dueAt.getTime() - offsetMinutes * 60_000);
}

export async function runReminderScheduler(dependencies: SchedulerDependencies): Promise<SchedulerResult> {
  const candidates = await dependencies.repository.listDueSmsCandidates(dependencies.now);
  const result: SchedulerResult = { scanned: candidates.length, sent: 0, skipped: 0, failed: 0 };

  for (const reminder of candidates) {
    for (const offset of reminder.reminderOffsets) {
      const scheduledAt = scheduledTimeForOffset(reminder.dueAt, offset);
      if (scheduledAt > dependencies.now) continue;
      const existingLog = await dependencies.repository.findDeliveryLog(reminder.id, "sms", scheduledAt);
      if (existingLog && (existingLog.status === "sent" || existingLog.status === "pending")) {
        result.skipped += 1;
        continue;
      }
      try {
        const sendResult = await dependencies.smsAdapter.sendReminder({ reminder, scheduledAt });
        await dependencies.repository.createDeliveryLog({
          reminderId: reminder.id,
          channel: "sms",
          status: "sent",
          scheduledAt,
          sentAt: dependencies.now,
          providerMessageId: sendResult.providerMessageId,
          errorMessage: null,
          attemptCount: 1
        });
        result.sent += 1;
      } catch (error) {
        await dependencies.repository.createDeliveryLog({
          reminderId: reminder.id,
          channel: "sms",
          status: "failed",
          scheduledAt,
          sentAt: null,
          providerMessageId: null,
          errorMessage: error instanceof Error ? error.message : "Unknown SMS adapter error",
          attemptCount: 1
        });
        result.failed += 1;
      }
    }
  }

  return result;
}
